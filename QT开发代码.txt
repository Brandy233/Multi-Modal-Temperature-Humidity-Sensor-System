// 私有辅助函数，处理数据插入的通用逻辑
bool sql::insertDataIntoDatabase(double temperature, double humidity)
{
    if (!db.isOpen()) {
        QMessageBox::critical(this, "错误", "数据库未连接！");
        return false;
    }
    if (!model) {
        QMessageBox::critical(this, "错误", "模型未初始化！");
        return false;
    }

    // 创建一条空记录
    QSqlRecord record = model->record();
    // 设置字段值
    record.setValue("temperature", temperature);
    record.setValue("humidity", humidity);
    // timestamp 字段设置为 DEFAULT CURRENT_TIMESTAMP，数据库会自动填充

    // 插入到模型（-1 表示插入到末尾）
    if (model->insertRecord(-1, record)) {
        // 提交到数据库
        if (model->submitAll()) {
            model->select(); // 刷新表格显示
            return true;
        } else {
            qDebug() << "提交失败：" << model->lastError().text();
            model->revertAll(); // 回滚，避免模型数据和数据库不一致
            return false;
        }
    } else {
        qDebug() << "插入记录失败：" << model->lastError().text();
        return false;
    }
}

// 公共槽函数，用于接收传感器数据并存入数据库
void sql::addSensorData(double temperature, double humidity)
{
    if (insertDataIntoDatabase(temperature, humidity)) {
        qDebug() << "传感器数据自动添加成功: 温度=" << temperature << ", 湿度=" << humidity;
    } else {
        qDebug() << "传感器数据自动添加失败: 温度=" << temperature << ", 湿度=" << humidity;
    }
}

void sql::on_pushButton_AddData_clicked()
{
    // 1. 获取用户输入
     QString tempStr = ui->lineEdit_Temp->text().trimmed();   // 温度
     QString humiStr = ui->lineEdit_chill->text().trimmed(); // 湿度

     // 2. 简单校验：不能为空
     if (tempStr.isEmpty() || humiStr.isEmpty()) {
         QMessageBox::warning(this, "警告", "温度或湿度不能为空！");
         return;
     }

     double temperature = tempStr.toDouble();
     double humidity = humiStr.toDouble();

     // 调用通用插入函数
     if (insertDataIntoDatabase(temperature, humidity)) {
         QMessageBox::information(this, "成功", "手动数据添加成功！");
         // 清空输入框
         ui->lineEdit_Temp->clear();
         ui->lineEdit_chill->clear();
     } else {
         QMessageBox::critical(this, "错误", "手动数据添加失败！请查看日志。");
     }


}




void sql::on_pushButton_Delete_clicked()
{

    // 1. 获取表格选中的行
        QModelIndexList selectedRows = ui->tableView_SQL->selectionModel()->selectedRows();
        if (selectedRows.isEmpty()) {
            QMessageBox::warning(this, "提示", "请先选中要删除的行！");
            return;
        }

        // 2. 弹出确认框（避免误删）
        if (QMessageBox::question(this, "确认删除",
            "是否删除选中的 " + QString::number(selectedRows.size()) + " 条数据？",
            QMessageBox::Yes | QMessageBox::No) != QMessageBox::Yes) {
            return;
        }

        // 3. 从模型中删除行（注意：要按行号从大到小删，避免索引混乱）
        //    倒序删除：防止删除一行后，后面行的索引变化
        for (int i = selectedRows.size() - 1; i >= 0; --i) {
            int row = selectedRows.at(i).row();
            model->removeRow(row);
        }

        // 4. 提交删除到数据库
        if (model->submitAll()) {
            QMessageBox::information(this, "成功", "数据删除成功！");
            model->select(); // 刷新表格
        } else {
            QMessageBox::critical(this, "错误", "删除失败：" + model->lastError().text());
            model->revertAll(); // 回滚，恢复模型数
        }
}

#include "widget.h"
#include "ui_widget.h"

#include <QTcpSocket>

Widget::Widget(QWidget *parent)
    : QWidget(parent)
    , ui(new Ui::Widget)
{
    ui->setupUi(this);
    this->setLayout(ui->verticalLayout_2);
    client = new QTcpSocket (this);
    connect(client , SIGNAL(readyRead()), this , SLOT(on_mread_Data()));
    ui->pushButton_stop->setEnabled(false);
}

Widget::~Widget()
{
    delete ui;
}


void Widget::on_pushButton_Connect_clicked()
{
    //    virtual void
    //    connectToHost(const QHostAddress &address, quint16 port, QIODevice::OpenMode openMode = ReadWrite)
    client->connectToHost(ui->lineEdit_IP->text() ,ui->lineEdit_port->text().toInt());
    connect(client , SIGNAL(connected()) , this , SLOT(on_connected()));
    connect(client , SIGNAL(error(QAbstractSocket::SocketError)) , this ,SLOT(on_Error(QAbstractSocket::SocketError)));
}

void Widget::on_connected()
{

        ui->textEdit_Rec->append("连接成功");
        ui->pushButton_Connect->setEnabled(false);
        ui->lineEdit_IP->setEnabled(false);
        ui->lineEdit_port->setEnabled(false);
        ui->pushButton_stop->setEnabled(true);

}

void Widget::on_Error(QAbstractSocket::SocketError error){
    qDebug() << "连接错误"  << error;
}

void Widget::on_mread_Data(){
    QString line = QString::fromUtf8( client->readAll());
        QStringList list = line.split("/");
        ui->textEdit_Rec->append("温湿度：" + list[0] + "℃" + "/" + list[1] + "%");
        ui->textEdit_Rec->moveCursor(QTextCursor::End);
        ui->textEdit_Rec->ensureCursorVisible();
}

void Widget::on_pushButton_stop_clicked()
{
    client->close();
    ui->textEdit_Rec->append("成功断开");
    ui->pushButton_Connect->setEnabled(true);
    ui->lineEdit_IP->setEnabled(true);
    ui->lineEdit_port->setEnabled(true);
    ui->pushButton_stop->setEnabled(false);

}

void Widget::on_pushButton_SQL_clicked()
{
    emit emit_show();
    this->hide();
}

void Widget::return_meau()
{
    this->show();
}
